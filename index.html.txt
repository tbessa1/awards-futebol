  <script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
    import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
    import { getFirestore, doc, setDoc, getDoc, collection, onSnapshot, addDoc, updateDoc, deleteDoc, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

    // Configuração do Firebase
    const firebaseConfig = JSON.parse(__firebase_config);
    const app = initializeApp(firebaseConfig);
    const auth = getAuth(app);
    const db = getFirestore(app);
    const appId = typeof __app_id !== 'undefined' ? __app_id : 'pinto-basto-awards';

    // Estado da App
    window.appState = {
        players: [],
        games: [],
        votes: [],
        currentUser: null,
        currentRankingTab: 'all',
        db: db,
        auth: auth,
        appId: appId
    };

    // Inicialização de Autenticação (REGRA 3)
    const initAuth = async () => {
        try {
            if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                await signInWithCustomToken(auth, __initial_auth_token);
            } else {
                await signInAnonymously(auth);
            }
        } catch (error) {
            console.error("Erro na autenticação:", error);
        }
    };

    onAuthStateChanged(auth, (user) => {
        if (user) {
            window.appState.user = user;
            setupRealtimeListeners();
        } else {
            initAuth();
        }
    });

    // Listeners em Tempo Real (REGRA 1 e 2)
    function setupRealtimeListeners() {
        const publicPath = ['artifacts', appId, 'public', 'data'];
        
        // Atletas
        onSnapshot(collection(db, ...publicPath, 'players'), (snapshot) => {
            window.appState.players = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
            updateActiveUI();
        }, (err) => console.error("Erro atletas:", err));

        // Jogos
        onSnapshot(collection(db, ...publicPath, 'games'), (snapshot) => {
            window.appState.games = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
            updateActiveUI();
        }, (err) => console.error("Erro jogos:", err));

        // Votos
        onSnapshot(collection(db, ...publicPath, 'votes'), (snapshot) => {
            window.appState.votes = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
            updateActiveUI();
        }, (err) => console.error("Erro votos:", err));
    }

    // Funções Globais expostas para o HTML
    window.savePlayerToCloud = async (playerData, id = null) => {
        const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'players');
        if (id) {
            await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'players', id), playerData);
        } else {
            await addDoc(colRef, playerData);
        }
    };

    window.deletePlayerFromCloud = async (id) => {
        await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'players', id));
    };

    window.saveGameToCloud = async (gameData, id = null) => {
        const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'games');
        if (id) {
            await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'games', id), gameData);
        } else {
            await addDoc(colRef, gameData);
        }
    };

    window.deleteGameFromCloud = async (id) => {
        await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'games', id));
    };

    window.submitVoteToCloud = async (voteData) => {
        // Encontra voto existente para atualizar ou cria novo
        const existing = window.appState.votes.find(v => 
            v.gameId === voteData.gameId && 
            v.voterId === voteData.voterId && 
            v.category === voteData.category && 
            v.targetId === voteData.targetId
        );

        if (existing) {
            await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'votes', existing.id), { score: voteData.score });
        } else {
            await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'votes'), voteData);
        }
    };

    // Inicialização da UI após o carregamento do Firebase
    window.addEventListener('load', () => {
        setTimeout(() => {
            document.getElementById('loading-screen').style.opacity = '0';
            setTimeout(() => {
                document.getElementById('loading-screen').style.display = 'none';
                document.getElementById('app').classList.remove('hidden');
            }, 500);
        }, 1500);
    });

  <script>
    const ADMIN_PASS = "admin123";
    const CATEGORIES = [
      { id: 'perf', name: 'Performance', icon: '⭐', type: 'positive' },
      { id: 'def', name: 'Defesa', icon: '🛡️', type: 'positive' },
      { id: 'gol', name: 'Golo', icon: '⚔️', type: 'positive' },
      { id: 'temp', name: 'Atitude', icon: '😤', type: 'positive' },
      { id: 'bat', name: 'Batota', icon: '⚠️', type: 'negative' }
    ];

    let tempPhoto = "";

    function updateActiveUI() {
      renderLoginOptions();
      renderResultsUI();
      if (window.appState.currentUser === 'admin') renderAdminDashboard();
      if (window.appState.currentUser && typeof window.appState.currentUser === 'object') renderPlayerDashboard();
    }

    function renderTabs() {
        const cont = document.getElementById('category-tabs');
        CATEGORIES.forEach(cat => {
            const btn = document.createElement('button');
            btn.className = 'category-tab px-4 py-2 rounded-xl text-[10px] font-black uppercase tracking-widest border border-white/10 bg-white/5';
            btn.textContent = cat.name;
            btn.onclick = () => changeRankingTab(cat.id);
            btn.id = `tab-${cat.id}`;
            cont.appendChild(btn);
        });
    }

    function changeRankingTab(tabId) {
        window.appState.currentRankingTab = tabId;
        document.querySelectorAll('.category-tab').forEach(t => t.classList.remove('active'));
        document.getElementById(`tab-${tabId}`).classList.add('active');
        renderResultsUI();
    }

    // Login Actions
    function loginAsAdmin() {
      if (document.getElementById('admin-password').value === ADMIN_PASS) {
        window.appState.currentUser = 'admin'; showView('admin');
      } else { alert('Chave incorreta.'); }
    }

    function loginAsPlayer() {
      const id = document.getElementById('player-select').value;
      if (!id) return;
      window.appState.currentUser = window.appState.players.find(x => x.id === id);
      if (window.appState.currentUser) showView('player');
    }

    function viewAsGuest() {
      window.appState.currentUser = 'guest';
      showView('guest');
    }

    function logout() {
      window.appState.currentUser = null;
      document.getElementById('main-app').classList.add('hidden');
      document.getElementById('login-screen').classList.remove('hidden');
      document.getElementById('ranking-section').classList.add('hidden');
    }

    function showView(role) {
      document.getElementById('login-screen').classList.add('hidden');
      document.getElementById('main-app').classList.remove('hidden');
      document.getElementById('ranking-section').classList.remove('hidden');
      const badge = document.getElementById('user-badge');
      const guestNotice = document.getElementById('guest-notice');
      
      badge.classList.remove('hidden');
      guestNotice.classList.add('hidden');

      if (role === 'admin') {
        badge.textContent = 'DIREÇÃO';
        document.getElementById('admin-dashboard').classList.remove('hidden');
        document.getElementById('player-dashboard').classList.add('hidden');
        renderAdminDashboard();
      } else if (role === 'guest') {
        badge.textContent = 'CONVIDADO';
        document.getElementById('admin-dashboard').classList.add('hidden');
        document.getElementById('player-dashboard').classList.add('hidden');
        guestNotice.classList.remove('hidden');
      } else {
        badge.textContent = window.appState.currentUser.name.split(' ')[0];
        document.getElementById('player-dashboard').classList.remove('hidden');

    // Dashboard Rendering
    function renderLoginOptions() {
      const select = document.getElementById('player-select');
      if (!select) return;
      const currentVal = select.value;
      select.innerHTML = '<option value="">Escolhe o teu nome...</option>';
      window.appState.players.sort((a,b) => a.name.localeCompare(b.name)).forEach(p => {
        const opt = document.createElement('option');
        opt.value = p.id; opt.textContent = p.name; select.appendChild(opt);
      });
      select.value = currentVal;
    }

    function renderResultsUI() {
      const { players, votes, currentRankingTab } = window.appState;
      const allStats = players.map(p => {
            const playerVotes = votes.filter(v => v.targetId === p.id);
            const catPoints = {};
            CATEGORIES.forEach(cat => {
                const scores = playerVotes.filter(v => v.category === cat.id).map(v => v.score);
                catPoints[cat.id] = scores.reduce((a, b) => a + b, 0);
            });
            const total = (catPoints['perf'] || 0) + (catPoints['def'] || 0) + (catPoints['gol'] || 0) + (catPoints['temp'] || 0) - (catPoints['bat'] || 0);
            return { ...p, catPoints, total };
      });

      let ranking = [];
      if (currentRankingTab === 'all') {
          ranking = allStats.sort((a, b) => b.total - a.total);
      } else {
          ranking = allStats.sort((a, b) => b.catPoints[currentRankingTab] - a.catPoints[currentRankingTab]);
      }
      
      const podiumCont = document.getElementById('podium-container');
      if (ranking.length > 0) {
          const top3 = [];
          if (ranking[1]) top3.push(ranking[1]);
          if (ranking[0]) top3.push(ranking[0]);
          if (ranking[2]) top3.push(ranking[2]);

          podiumCont.innerHTML = top3.map(p => {
              const isFirst = (p.id === ranking[0].id);
              const pts = currentRankingTab === 'all' ? p.total : p.catPoints[currentRankingTab];
              return `
                <div class="flex flex-col items-center gap-4 ${isFirst ? 'mb-8' : 'mb-0'}">
                    <div class="relative">
                        <div class="${isFirst ? 'podium-1 w-36 h-36' : 'w-24 h-24 opacity-70'} rounded-[2rem] overflow-hidden border-2 border-white/10 bg-[#0f172a]">
                            ${p.photo ? `<img src="${p.photo}" class="w-full h-full object-cover">` : ''}
                        </div>
                    </div>
                    <div class="text-center">
                        <p class="font-black text-white text-[10px] uppercase">${p.name.split(' ')[0]}</p>
                        <p class="text-[9px] font-black text-blue-500 uppercase mt-1">${pts} PTS</p>
                    </div>
                </div>`;
          }).join('');
      }

      document.getElementById('results-table-body').innerHTML = ranking.map((p, idx) => {
        const pts = currentRankingTab === 'all' ? p.total : p.catPoints[currentRankingTab];
        return `
        <tr class="hover:bg-white/5 transition-colors">
            <td class="px-6 py-5"><span class="w-7 h-7 rounded-lg ${idx === 0 ? 'bg-yellow-500' : 'bg-white/5'} flex items-center justify-center text-[9px] font-black text-white italic">${idx + 1}</span></td>
            <td class="px-6 py-5">
              <div class="flex items-center gap-3">
                <div class="w-8 h-8 rounded-lg overflow-hidden bg-white/5 border border-white/10">${p.photo ? `<img src="${p.photo}" class="w-full h-full object-cover">` : ''}</div>
                <div><span class="font-bold text-white text-xs block">${p.name}</span><span class="text-[7px] font-black text-slate-500 uppercase">${p.position}</span></div>
              </div>
            </td>
            ${CATEGORIES.map(cat => `<td class="px-4 py-5 text-center text-[10px] ${currentRankingTab === cat.id ? 'bg-blue-600/20 text-white' : 'text-slate-400'}">${p.catPoints[cat.id] || 0}</td>`).join('')}
            <td class="px-6 py-5 text-right font-black text-blue-500 bg-blue-500/5 text-sm">${pts}</td>
        </tr>`;
      }).join('');
    }

    function renderPlayerDashboard() {
      const myGames = window.appState.games.filter(g => g.playerIds.includes(window.appState.currentUser.id));
      document.getElementById('current-player-photo').innerHTML = window.appState.currentUser.photo ? `<img src="${window.appState.currentUser.photo}" class="w-full h-full object-cover">` : '';
      document.getElementById('player-welcome-name').textContent = window.appState.currentUser.name.split(' ')[0];
      
      const list = document.getElementById('voting-games-list');
      if (myGames.length === 0) {
        list.innerHTML = `<div class="p-10 text-center border-2 border-dashed border-white/5 rounded-[3rem] text-slate-700 text-[10px] font-black uppercase italic">Nenhuma convocatória disponível.</div>`;
        return;
      }

      list.innerHTML = myGames.map(g => {
        const others = window.appState.players.filter(p => p.id !== window.appState.currentUser.id && g.playerIds.includes(p.id));
        const currentVotes = window.appState.votes.filter(v => v.gameId === g.id && v.voterId === window.appState.currentUser.id).length;
        const requiredVotes = others.length * CATEGORIES.length;
        const isFinished = requiredVotes > 0 && currentVotes >= requiredVotes;
        
        return `
          <div class="p-6 bg-[#0f172a] rounded-[2.5rem] border border-white/5 flex flex-col md:flex-row justify-between items-center gap-4">
            <div class="flex-1">
              <h4 class="font-black text-white text-xl uppercase italic">${g.title}</h4>
              <p class="text-[8px] text-slate-600 font-black uppercase tracking-widest">${g.date}</p>
            </div>
            ${isFinished ? 
              `<div class="px-6 py-3 bg-green-500/10 text-green-500 rounded-xl font-black text-[9px] uppercase border border-green-500/20">Votado ✓</div>` :
              `<button onclick="openVoting('${g.id}')" class="px-8 py-3 bg-blue-600 text-white rounded-xl font-black text-[9px] uppercase">Votar</button>`
            }
          </div>`;
      }).reverse().join('');
    }

    function openVoting(gameId) {
      const game = window.appState.games.find(g => g.id === gameId);
      const others = window.appState.players.filter(p => p.id !== window.appState.currentUser.id && game.playerIds.includes(p.id));
      const content = document.getElementById('voting-modal-content');
      
      content.innerHTML = CATEGORIES.map(cat => `
        <div class="space-y-6">
          <div class="flex items-center gap-4 border-b border-white/5 pb-4">
              <span class="text-3xl">${cat.icon}</span>
              <h4 class="text-white font-black text-xl uppercase italic">${cat.name}</h4>
          </div>
          <div class="grid gap-4">
            ${others.map(p => {
              const currentV = window.appState.votes.find(v => v.gameId === gameId && v.voterId === window.appState.currentUser.id && v.category === cat.id && v.targetId === p.id);
              const score = currentV ? currentV.score : -1;
              return `
                <div class="bg-white/5 p-5 rounded-3xl flex flex-col lg:flex-row items-center justify-between gap-4 border border-white/5">
                  <span class="text-[10px] font-black text-white uppercase">${p.name}</span>
                  <div class="flex flex-wrap justify-center gap-1">
                    ${[0, 1, 2, 3, 4, 5].map(val => `
                      <button onclick="setScore('${gameId}','${cat.id}','${p.id}', ${val}, this)" 
                        class="rating-btn w-10 h-10 rounded-xl border border-white/10 font-black text-xs
                        ${score === val ? 'active' : 'bg-[#1e293b] text-slate-500'}">
                        ${val}
                      </button>
                    `).join('')}
                  </div>
                </div>`;
            }).join('')}
          </div>
        </div>`).join('');
      openModal('voting-modal');
    }

    // Database Write Actions
    async function setScore(gId, catId, tId, score, btn) {
        await window.submitVoteToCloud({
            gameId: gId,
            voterId: window.appState.currentUser.id,
            category: catId,
            targetId: tId,
            score: score
        });
        btn.parentElement.querySelectorAll('button').forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
    }

    async function savePlayer() {
      const id = document.getElementById('edit-player-id').value;
      const name = document.getElementById('new-player-name').value.trim();
      const pos = document.getElementById('new-player-position').value;
      if (!name) return;
      await window.savePlayerToCloud({ name, position: pos, photo: tempPhoto }, id || null);
      closeModal('player-modal');
    }

    async function deletePlayer(id) {
      if(!confirm('Remover definitivamente?')) return;
      await window.deletePlayerFromCloud(id);
    }

    async function saveGame() {
      const id = document.getElementById('edit-game-id').value;
      const title = document.getElementById('new-game-title').value;
      const date = document.getElementById('new-game-date').value;
      const pIds = Array.from(document.querySelectorAll('.player-checkbox:checked')).map(i => i.value);
      if (!title || pIds.length === 0) return;
      await window.saveGameToCloud({ title, date, playerIds: pIds }, id || null);
      closeModal('game-modal');
    }

    async function deleteGame(id) {
      if(!confirm('Eliminar jogo?')) return;
      await window.deleteGameFromCloud(id);
    }

    function handlePhotoUpload(input) {
      if (input.files?.[0]) {
        const reader = new FileReader();
        reader.onload = e => {
          tempPhoto = e.target.result;
          document.getElementById('photo-preview-container').innerHTML = `<img src="${tempPhoto}" class="w-full h-full object-cover">`;
        };
        reader.readAsDataURL(input.files[0]);
      }
    }
